  3 BetaJS.Browser.Loader = {

  5 	loadScript: function (url, callback, context) {
  6 		var executed = false;
  7 		var head = document.getElementsByTagName("head")[0];
  8 		var script = document.createElement("script");
  9 		script.src = url;
 10 		script.onload = script.onreadystatechange = function() {
 11 			if (!executed && (!this.readyState || this.readyState == "loaded" || this.readyState == "complete")) {
 12 				executed = true;
 13 				script.onload = script.onreadystatechange = null;
 14 				if (callback)
 15 					callback.apply(context || this, [url]);
 16 				// Does not work properly if we remove the script for some reason if it is used the second time !?
 17 				//head.removeChild(script);
 18 			}
 19 		};
 20 		head.appendChild(script);
 21 	},
 22 	
 23 	loadStyles: function (url, callback, context) {
 24 		var executed = false;
 25 		var head = document.getElementsByTagName("head")[0];
 26 		var style = document.createElement("link");
 27 		style.rel = "stylesheet";
 28 		style.href = url;
 29 		style.onload = style.onreadystatechange = function() {
 30 			if (!executed && (!this.readyState || this.readyState == "loaded" || this.readyState == "complete")) {
 31 				executed = true;
 32 				style.onload = style.onreadystatechange = null;
 33 				if (callback)
 34 					callback.apply(context || this, [url]);
 35 			}
 36 		};
 37 		head.appendChild(style);
 38 	},
 39 
 40 	inlineStyles: function (styles) {
 41 		BetaJS.$('<style>' + styles + "</style>").appendTo("head");
 42 	}
