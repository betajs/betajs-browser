  3 	traverseNext: function (node, skip_children) {
  4 		if ("get" in node)
  5 			node = node.get(0);
  6 		if (node.firstChild && !skip_children)
  7 			return BetaJS.$(node.firstChild);
  8 		if (!node.parentNode)
  9 			return null;
 10 		if (node.nextSibling)
 11 			return BetaJS.$(node.nextSibling);
 12 		return this.traverseNext(node.parentNode, true);
 13 	},

 15 	selectNode : function(node, offset) {
 16 		node = BetaJS.$(node).get(0);
 17 		var selection = null;
 18 		var range = null;
 19 		if (window.getSelection) {
 20 			selection = window.getSelection();
 21 			selection.removeAllRanges();
 22 			range = document.createRange();
 23 		} else if (document.selection) {
 24 			selection = document.selection;
 25 			range = selection.createRange();
 26 		}
 27 		if (offset) {
 28 			range.setStart(node, offset);
 29 			range.setEnd(node, offset);
 30 			selection.addRange(range);
 31 		} else {
 32 			range.selectNode(node);
 33 			selection.addRange(range);
 34 		}
 35 	},

 93 	selectionNonEmpty: function () {
 94 		var start = this.selectionStart();
 95 		var end = this.selectionEnd();
 96 		return start && end && start.get(0) && end.get(0) && (start.get(0) != end.get(0) || this.selectionStartOffset() != this.selectionEndOffset());
 97 	},

 99 	selectionContained: function (node) {
100 		return node.has(this.selectionStart()).length > 0 && node.has(this.selectionEnd()).length > 0;

103 	selectionNodes: function () {
104 		var result = [];
105 		var start = this.selectionStart();
106 		var end = this.selectionEnd();
107 		result.push(start);
108 		var current = start;
109 		while (current.get(0) != end.get(0)) {
110 			current = this.traverseNext(current);
111 			result.push(current);
112 		}
113 		return result;
114 	},

126 	remove_tag_from_parent_path: function (node, tag, context) {	
127 		tag = tag.toLowerCase();
128 		node = BetaJS.$(node);
129 		var parents = node.parents(context ? context + " " + tag : tag);
130 		for (var i = 0; i < parents.length; ++i) {
131 			var parent = parents.get(i);
132 			parent = BetaJS.$(parent);
133 			while (node.get(0) != parent.get(0)) {
134 				this.contentSiblings(node).wrap("<" + tag + "></" + tag + ">");
135 				node = node.parent();
136 			}
137 			parent.contents().unwrap();
138 		}

141 	selectionSplitOffsets: function () {
142 		var startOffset = this.selectionStartOffset();
143 		var endOffset = this.selectionEndOffset();
144 		var start = this.selectionStart();
145 		var end = this.selectionEnd();
146 		var single = start.get(0) == end.get(0);
147 		if (endOffset < end.get(0).wholeText.length) {
148 			var endElem = end.get(0);
149 			endElem.splitText(endOffset);
150 			end = BetaJS.$(endElem);
151 			if (single)
152 				start = end;
153 		}
154 		if (startOffset > 0) {
155 			start = BetaJS.$(start.get(0).splitText(startOffset));
156 			if (single)
157 				end = start;
158 		}
159 		this.selectRange(start, end);
160 	},

162 	selectRange: function (start_node, end_node, start_offset, end_offset) {
163 		start_node = BetaJS.$(start_node);
164 		end_node = BetaJS.$(end_node);
165 		var selection = null;
166 		var range = null;
167 		if (window.getSelection) {
168 			selection = window.getSelection();
169 			selection.removeAllRanges();
170 			range = document.createRange();
171 		} else if (document.selection) {
172 			selection = document.selection;
173 			range = selection.createRange();
174 		}
175 		range.setStart(start_node.get(0), start_offset || 0);
176 		range.setEnd(end_node.get(0), end_offset || end_node.get(0).data.length);
177 		selection.addRange(range);
178 	},

180 	splitNode: function (node, start_offset, end_offset) {
181 		node = BetaJS.$(node);
182 		start_offset = start_offset || 0;
183 		end_offset = end_offset || node.get(0).data.length;
184 		if (end_offset < node.get(0).data.length) {
185 			var elem = node.get(0);
186 			elem.splitText(end_offset);
187 			node = BetaJS.$(elem);
188 		}
189 		if (start_offset > 0) 
190 			node = BetaJS.$(node.get(0).splitText(start_offset));
191 		return node;
192 	}
