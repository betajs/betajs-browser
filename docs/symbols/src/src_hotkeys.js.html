 92 	keyCodeToCharacter: function (code) {
 93 		if (code == 188)
 94 			return ",";
 95 		else if (code == 190)
 96 			return ".";
 97 		return String.fromCharCode(code).toLowerCase();
 98 	},

100 	register: function (hotkey, callback, context, options) {
101 		options = BetaJS.Objs.extend({
102 			"type": "keyup",
103 			"propagate": false,
104 			"disable_in_input": false,
105 			"target": document,
106 			"keycode": false
107 		}, options);
108 		options.target = BetaJS.$(options.target);
109 		var keys = hotkey.toLowerCase().split("+");
110 		var func = function (e) {
111 			if (options.disable_in_input) {
112 				var element = e.target || e.srcElement || null;
113 				if (element && element.nodeType == 3)
114 					element = element.parentNode;
115 				if (element && (element.tagName == 'INPUT' || element.tagName == 'TEXTAREA'))
116 					return;
117 			}
118 			var code = e.keyCode || e.which || 0;
119 			var character = BetaJS.Browser.Hotkeys.keyCodeToCharacter(code);
120 			var kp = 0;
121 			var modifier_map = {};
122 			BetaJS.Objs.iter(BetaJS.Browser.Hotkeys.MODIFIERS, function (mod) {
123 				modifier_map[mod] = {
124 					pressed: e[mod + "Key"],
125 					wanted: false
126 				};
127 			}, this);
128 			BetaJS.Objs.iter(keys, function (key) {
129 				if (key in modifier_map) {
130 					modifier_map[key].wanted = true;
131 					kp++;
132 				} else if (key.length > 1) {
133 					if (BetaJS.Browser.Hotkeys.SPECIAL_KEYS[key] == code)
134 						kp++;
135 				} else if (options.keycode) {
136 					if (options.keycode == code)
137 						kp++;
138 				} else if (character == key || (e.shiftKey && BetaJS.Browser.Hotkeys.SHIFT_NUMS[character] == key)) {
139 					kp++;
140 				}
141 			}, this);
142 			if (kp == keys.length && BetaJS.Objs.all(modifier_map, function (data) { return data.wanted == data.pressed; })) {
143 				callback.apply(context || this);
144 				if (!options.propagate)
145 					e.preventDefault();
146 			}
147 		};
148 		options.target.on(options.type, func);
149 		return {
150 			target: options.target,
151 			type: options.type,
152 			func: func
153 		};
154 	},
155 	
156 	unregister: function (handle) {
157 		handle.target.off(handle.type, handle.func);
158 	}
