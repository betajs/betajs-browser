 96 	 * @return either null if nothing applicable and valid could be matched or an associative array with params and routing object as attributes.

112 				BetaJS.Objs.iter(obj.valid, function (s) {

156 		this.trigger("navigate", route);

158 		if (result === null) {

160 			return false;
161 		}
162 		this.trigger("navigate-success", result.object, result.params);
